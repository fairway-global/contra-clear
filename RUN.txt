================================================================================
  CONTRA OTC DEMO — Run Guide
================================================================================

ARCHITECTURE
------------
  Port   | Service              | Type          | Purpose
  -------|----------------------|---------------|----------------------------------
  5432   | PostgreSQL           | Homebrew      | Contra channel state + indexer DB
  18899  | Solana Validator     | Native binary | Local "mainnet" with escrow/withdraw programs
  8900   | Contra Write Node    | Native binary | Channel transaction processing
  8901   | Contra Read Node     | Native binary | Channel balance/state queries
  8899   | Contra Gateway       | Native binary | Routes RPC to write/read nodes
  —      | Indexer-Solana       | Native binary | Watches validator for escrow deposits
  —      | Indexer-Contra       | Native binary | Watches channel for withdrawal burns
  —      | Operator-Solana      | Native binary | Mints tokens on channel after deposits
  —      | Operator-Contra      | Native binary | Releases tokens from escrow after burns
  3001   | OTC API Server       | Node.js       | RFQ, trades, balances, auth
  3002   | OTC WebSocket        | Node.js       | Real-time push events
  5173   | OTC Frontend         | Vite dev      | React trading UI


PREREQUISITES
-------------
  - PostgreSQL running locally (brew services start postgresql@14)
  - Rust toolchain installed
  - Solana CLI 2.2.19 (for program builds: agave-install init 2.2.19)
  - Node.js 22+, pnpm
  - Contra repo cloned at ./contra/ with programs built


FIRST-TIME SETUP
----------------
  # 1. Create databases (once)
  psql postgres -c "CREATE USER contra WITH PASSWORD 'contra_password' SUPERUSER;" 2>/dev/null
  psql postgres -c "CREATE DATABASE contra OWNER contra;" 2>/dev/null
  psql postgres -c "CREATE DATABASE indexer OWNER contra;" 2>/dev/null

  # 2. Build Contra (if not done)
  cd contra
  agave-install init 2.2.19
  cd contra-withdraw-program/program && cargo-build-sbf && cd ../..
  cd contra-escrow-program/program && cargo-build-sbf && cd ../..
  cargo build --release --bin node --bin gateway --bin indexer
  cd ..

  # 3. Install OTC dependencies
  cd otc-server && npm install && cd ..
  cd otc-frontend && npm install && cd ..

  # 4. Run setup script (creates token mints, test wallets, escrow instance)
  bash scripts/setup-demo.sh
  node otc-server/create-escrow-instance.mjs   # from otc-server/ dir

  # 5. Configure .env files (setup script does this automatically)
  #    - otc-server/.env  (DEMO_TOKEN_MINTS, ESCROW_INSTANCE_ID, etc.)
  #    - otc-frontend/.env (VITE_DEMO_TOKEN_MINTS)


START ALL SERVICES (in order)
-----------------------------
  Run each command in a separate terminal, or use the one-liner at the bottom.

  # Terminal 1: Solana Validator
  solana-test-validator \
    --rpc-port 18899 --faucet-port 19900 --reset \
    --bpf-program GokvZqD2yP696rzNBNbQvcZ4VsLW7jNvFXU1kW9m7k83 contra/target/deploy/contra_escrow_program.so \
    --bpf-program J231K9UEpS4y4KAPwGc4gsMNCjKFRMYcQBcjVW7vBhVi contra/target/deploy/contra_withdraw_program.so \
    --ledger /tmp/contra-test-ledger

  # Terminal 2: Contra Write Node
  cd contra && \
  CONTRA_PORT=8900 \
  CONTRA_ACCOUNTSDB_CONNECTION_URL="postgres://contra:contra_password@localhost:5432/contra" \
  CONTRA_ENABLE_READ=false CONTRA_SIGVERIFY_QUEUE_SIZE=10000 CONTRA_SIGVERIFY_WORKERS=4 \
  CONTRA_MAX_CONNECTIONS=100 CONTRA_MAX_TX_PER_BATCH=64 CONTRA_MODE=write \
  CONTRA_ADMIN_KEYS=$(cat ../.env.demo 2>/dev/null | grep ADMIN_PUBKEY | cut -d= -f2) \
  RUST_LOG=info ./target/release/node

  # Terminal 3: Contra Read Node
  cd contra && \
  CONTRA_PORT=8901 \
  CONTRA_ACCOUNTSDB_CONNECTION_URL="postgres://contra:contra_password@localhost:5432/contra" \
  CONTRA_MAX_CONNECTIONS=100 CONTRA_ENABLE_WRITE=false CONTRA_ENABLE_READ=true \
  CONTRA_MODE=read RUST_LOG=info ./target/release/node

  # Terminal 4: Contra Gateway
  cd contra && \
  GATEWAY_PORT=8899 GATEWAY_WRITE_URL=http://localhost:8900 \
  GATEWAY_READ_URL=http://localhost:8901 RUST_LOG=info ./target/release/gateway

  # Terminal 5: Indexer-Solana (watches validator for deposits)
  cd contra && \
  DATABASE_URL="postgres://contra:contra_password@localhost:5432/indexer" \
  RUST_LOG=info,contra_indexer=debug METRICS_PORT=9100 \
  ./target/release/indexer --config indexer/config/local/indexer-solana.toml -v indexer

  # Terminal 6: Indexer-Contra (watches channel for withdrawals)
  cd contra && \
  DATABASE_URL="postgres://contra:contra_password@localhost:5432/indexer" \
  RUST_LOG=info,contra_indexer=debug METRICS_PORT=9101 \
  ./target/release/indexer --config indexer/config/local/indexer-contra.toml -v indexer

  # Terminal 7: Operator-Solana (mints on channel after deposits)
  cd contra && \
  DATABASE_URL="postgres://contra:contra_password@localhost:5432/indexer" \
  ADMIN_SIGNER=memory ADMIN_PRIVATE_KEY=$(cat ../.env.demo 2>/dev/null | grep ADMIN_PRIVATE | cut -d= -f2 || echo "SET_ME") \
  OPERATOR_SIGNER=memory OPERATOR_PRIVATE_KEY=$(cat ../.env.demo 2>/dev/null | grep ADMIN_PRIVATE | cut -d= -f2 || echo "SET_ME") \
  RUST_LOG=info,contra_indexer=debug METRICS_PORT=9102 \
  ./target/release/indexer --config indexer/config/local/operator-solana.toml -v operator

  # Terminal 8: Operator-Contra (releases from escrow after burns)
  cd contra && \
  DATABASE_URL="postgres://contra:contra_password@localhost:5432/indexer" \
  ADMIN_SIGNER=memory ADMIN_PRIVATE_KEY=$(cat ../.env.demo 2>/dev/null | grep ADMIN_PRIVATE | cut -d= -f2 || echo "SET_ME") \
  OPERATOR_SIGNER=memory OPERATOR_PRIVATE_KEY=$(cat ../.env.demo 2>/dev/null | grep ADMIN_PRIVATE | cut -d= -f2 || echo "SET_ME") \
  RUST_LOG=info,contra_indexer=debug METRICS_PORT=9103 \
  ./target/release/indexer --config indexer/config/local/operator-contra.toml -v operator

  # Terminal 9: OTC API Server
  cd otc-server && npm run dev

  # Terminal 10: OTC Frontend
  cd otc-frontend && npm run dev

STOP ALL SERVICES
-----------------
  lsof -ti:3001 | xargs kill -9 2>/dev/null   # OTC server
  lsof -ti:3002 | xargs kill -9 2>/dev/null   # WebSocket
  lsof -ti:5173 | xargs kill -9 2>/dev/null   # Frontend
  lsof -ti:8899 | xargs kill -9 2>/dev/null   # Gateway
  lsof -ti:8900 | xargs kill -9 2>/dev/null   # Write node
  lsof -ti:8901 | xargs kill -9 2>/dev/null   # Read node
  lsof -ti:18899 | xargs kill -9 2>/dev/null  # Validator
  pkill -f "target/release/indexer" 2>/dev/null # Indexers + Operators


CHECK SERVICE HEALTH
--------------------
  curl -s http://localhost:18899 -X POST -H "Content-Type: application/json" \
    -d '{"jsonrpc":"2.0","id":1,"method":"getSlot"}' && echo ""       # Validator
  curl -s http://localhost:8899 -X POST -H "Content-Type: application/json" \
    -d '{"jsonrpc":"2.0","id":1,"method":"getSlot"}' && echo ""       # Gateway
  curl -s http://localhost:3001/health && echo ""                      # OTC Server
  curl -s -o /dev/null -w "Frontend: %{http_code}\n" http://localhost:5173  # Frontend

CLEAN STATE (full reset)
------------------------
  # Stop all services first, then:
  psql postgres -c "DROP DATABASE IF EXISTS contra;"
  psql postgres -c "DROP DATABASE IF EXISTS indexer;"
  psql postgres -c "CREATE DATABASE contra OWNER contra;"
  psql postgres -c "CREATE DATABASE indexer OWNER contra;"
  rm -f otc-server/otc.db otc-server/otc.db-wal otc-server/otc.db-shm
  rm -rf /tmp/contra-test-ledger
  rm -rf keypairs/
  # Then re-run FIRST-TIME SETUP steps 4-5


NOTES
-----
  - The validator resets on every start (--reset flag). All on-chain state is ephemeral.
  - After validator reset, you must re-run setup-demo.sh and create-escrow-instance.mjs
  - Phantom wallet must be set to Custom RPC http://localhost:18899 for deposits
  - The Contra channel (gateway port 8899) is separate from the validator (18899)
  - Admin key: check .env.demo for ADMIN_PUBKEY and keypairs/admin.json
  - Trade settlement is OTC-desk level (database), not on-channel transfers
  - Deposits and withdrawals ARE real on-chain Contra operations
